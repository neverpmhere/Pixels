function processFileContent(content) { var rows = content.split('\n'); var tableBody = document.getElementById('tableBody'); for (var i = 0; i < rows.length; i++) { var rowValues = rows[i].split(' '); var row = document.createElement('tr'); for (var j = 0; j < rowValues.length; j++) { var cell = document.createElement('td'); var value = parseFloat(rowValues[j].replace(/\s/g, '')); // Добавление разделителя для тысяч и больших чисел var formattedValue = value.toLocaleString(); // Добавление "+" в начало всех цифр в третьем столбце if (j === 2) { formattedValue = '+' + formattedValue; } cell.textContent = formattedValue; row.appendChild(cell); } tableBody.appendChild(row); } } // Функция для чтения файла function readFile() { var xhr = new XMLHttpRequest(); xhr.open('GET', 'tables.txt', true); xhr.onreadystatechange = function () { if (xhr.readyState == 4 && xhr.status == 200) { processFileContent(xhr.responseText); } }; xhr.send(); } // Вызов функции чтения файла при загрузке страницы window.onload = function () { readFile(); };
